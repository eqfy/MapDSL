        function createBlock(nwPosition, blockSize, northStreetName, eastStreetName, southStreetName, westStreetName) {
                variable nePosition = (nwPosition.x + blockSize, nwPosition.y);
                variable sePosition = (nwPosition.x + blockSize, nwPosition.y - blockSize);
                variable swPosition = (nwPosition.x - blockSize, nwPosition.y);
                return
                    {
                        type: "street",
                        name: northStreetName,
                        shape: "straight",
                        startPosition: nwPosition,
                        endPosition: nePosition
                    }
                    {
                        type: "street",
                        name: eastStreetName,
                        shape: "straight",
                        startPosition: nePosition,
                        endPosition: sePosition
                    }
                    {
                        type: "street",
                        name: southStreetName,
                        shape: "straight",
                        startPosition: sePosition,
                        endPosition: swPosition
                    }
                    {
                        type: "street",
                        name: westStreetName,
                        shape: "straight",
                        startPosition: swPosition,
                        endPosition: nwPosition
                    }
        }

        function createTrafficLight(trafficLightPosition) = {
                return

                        position: trafficLightPosition
                    }
